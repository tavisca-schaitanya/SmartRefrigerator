pipeline {
  agent any
  
  stages {
      stage('Build') {
          steps {
                sh 'dotnet build SmartRefrigerator.sln -p:configuration=release -v:n'
          }
      }
      stage('Test') {
          steps {
                sh 'dotnet test'
          }
     }
     stage('Publish') {
          steps {
                sh 'dotnet publish'
          }
     }
   }
    post { 
        success { 
            stage('Release') {
              steps {
                  dir('SmartRefrigeratorArchive') {
                      archiveArtifacts artifacts: '**', fingerprint: true
                  }
             }
          }
        }
    }
 }
